\begin{equation*} \frac{dy}{y^2} = (6 + t)\,dt, \end{equation*}
so
\begin{equation*} \int \frac{dy}{y^2} = \int (6 + t)\,dt, \end{equation*}
and
\begin{equation*} -\frac{1}{y} = {1\over2}(6 + t)^2 + C, \end{equation*}
giving
\begin{equation*} y=-\frac{2}{(6 + t)^2 + A}, \end{equation*}
where \(A = 2C\text{.}\) Since \(y=4\) when \(t=1\text{,}\) we have
\begin{equation*} 4 = -\frac{2}{7^2 + A},\quad \hbox{ so } \quad 4 A + 196 = -2, \end{equation*}
and \(A = -{{99\over2}}\text{.}\) Thus
\begin{equation*} y = -\frac{2}{(6 + t)^2 - {{99\over2}}}. \end{equation*}
