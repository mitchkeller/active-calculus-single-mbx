\begin{align*} f'(1) &= \lim_{h \to 0} \frac{f(1+h)-f(1)}{h}\\ &= \lim_{h \to 0} \frac{\frac{1}{1+h} - 1}{h}\\ &= \lim_{h \to 0} \frac{\frac{1}{1+h} - 1}{h} \left(\frac{1+h}{1+h} \right)\\ &= \lim_{h \to 0} \frac{1-(1+h)}{h(1+h)}\\ &= \lim_{h \to 0} \frac{-h)}{h(1+h)}\\ &= \lim_{h \to 0} -\frac{1}{1+h}\\ &= -1 \end{align*}
