\begin{equation*} T_4(x) = 0 + 1 (x-1) - \frac{1}{2} (x-1)^2 + \frac{(-2)(-1)}{3!}(x-1)^3 + \frac{(-3)(-2)(-1)}{4!}(x-1)^4\text{,} \end{equation*}
