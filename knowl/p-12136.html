\begin{equation*} x = {\textstyle \frac{1}{\theta}}\cos(\theta) \;\text{ and }\; y = {\textstyle \frac{1}{\theta}}\sin(\theta). \end{equation*}
So
\begin{equation*} \frac{dy}{dx}= \frac{dy/d\theta}{dx/d\theta}=\frac{\frac{d}{d\theta}\left(\frac{1}{\theta}\sin(\theta)\right)}{\frac{d}{d\theta}\left(\frac{1}{\theta}\cos(\theta)\right)} \end{equation*}
\begin{equation*} =\frac{-\frac{1}{\theta^2}\sin(\theta)+\frac{1}{\theta}\cos(\theta)} {-\frac{1}{\theta^2}\cos(\theta)-\frac{1}{\theta}\sin(\theta)} \end{equation*}
and by multiplying the top and bottom by \(-\theta^2\text{,}\) this simplies to
\begin{equation*} \frac{dy}{dx}= \frac{\sin(\theta)-\theta\cos(\theta)}{\cos(\theta)+\theta\sin(\theta)}\, . \end{equation*}
Therefore
\begin{equation*} \frac{dy}{dx}\Big|_{\theta= \pi}=\frac{\sin(\pi)-\pi\cos(\pi)}{\cos(\pi)+\pi\sin(\pi)} \end{equation*}
\begin{equation*} =\frac{0-\pi(-1)}{(-1)+\pi(0)}=\frac{\pi}{-1}=-\pi\, . \end{equation*}
