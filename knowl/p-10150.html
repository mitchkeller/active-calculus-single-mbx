\begin{equation*} \begin{aligned} \int \sin^{6}\!\left(8x\right)\cos^{3}\!\left(8x\right) \, dx \amp = \int \sin^{6} (8 x) \cos^2(8 x) \cos(8 x) \, dx \\\\ \amp = \int \sin^{6} (8 x) \bigg(1-\sin^2(8 x)\bigg) \cos(8 x) \, dx \\\\ \amp = \int u^{6} (1-u^2) \cos(8 x) \, \frac{ du}{8 \cos (8 x)} \\\\ \amp = \frac{1}{8}\int u^{6} (1-u^2) \, du \\\\ \amp = \frac{1}{8}\int u^{6} -u^{8} \, du \\\\ \amp = \frac{1}{8} \left[ \frac{u^{7}}{7} - \frac{u^{9}}{9} \right]+C \\\\ \amp = \frac{1}{8} \left[ \frac{\sin^{7}(8 x) }{7} - \frac{\sin^{9}(8 x)}{9} \right]+C \\\\ \end{aligned} \end{equation*}
