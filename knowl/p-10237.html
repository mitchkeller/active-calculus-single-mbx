Then let \(u= \sec(-15 x)\text{,}\) which means that \(du= -15 \sec(-15 x)\tan(-15 x) \, dx\text{,}\) and thus \(dx= \frac{du}{-15\sec(-15 x)\tan(-15 x)}\text{.}\)
\begin{equation*} \begin{aligned} \int 7\tan^{3}\!\left(-15x\right)\sec^{3}\!\left(-15x\right) \, dx \amp = 7 \int \tan^2(-15 x) \sec^2(-15 x) \sec(-15 x) \tan(-15 x) \, dx \\\\ \amp = 7 \int \bigg( \sec^2(-15 x)-1 \bigg) \sec^2(-15 x) \sec(-15 x) \tan(-15 x) \, dx \\\\ \amp = 7 \int \bigg( u^2-1 \bigg) u^2 \sec(-15 x) \tan(-15 x) \frac{du}{-15\sec(-15 x)\tan(-15 x)} \\\\ \amp = \frac{7}{-15} \int \bigg( u^2-1 \bigg) u^2 \, du \\\\ \amp = \frac{7}{-15} \int \bigg( u^4-u^2 \bigg) \, du \\\\ \amp = \frac{7}{-15} \bigg[\frac{ u^5}{5}-\frac{u^3}{3}\bigg]+C \\\\ \amp = \frac{7}{-15} \bigg[\frac{ \sec^5(-15 x)}{5}-\frac{\sec^3(-15 x)}{3}\bigg]+C \end{aligned} \end{equation*}
