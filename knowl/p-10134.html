\begin{equation*} \begin{aligned} \int \sin^{2}\!\left(-6x\right)\cos^{3}\!\left(-6x\right) \, dx \amp = \int \sin^{2} (-6 x) \cos^2(-6 x) \cos(-6 x) \, dx \\\\ \amp = \int \sin^{2} (-6 x) \bigg(1-\sin^2(-6 x)\bigg) \cos(-6 x) \, dx \\\\ \amp = \int u^{2} (1-u^2) \cos(-6 x) \, \frac{ du}{-6 \cos (-6 x)} \\\\ \amp = \frac{1}{-6}\int u^{2} (1-u^2) \, du \\\\ \amp = \frac{1}{-6}\int u^{2} -u^{4} \, du \\\\ \amp = \frac{1}{-6} \left[ \frac{u^{3}}{3} - \frac{u^{5}}{5} \right]+C \\\\ \amp = \frac{1}{-6} \left[ \frac{\sin^{3}(-6 x) }{3} - \frac{\sin^{5}(-6 x)}{5} \right]+C \\\\ \end{aligned} \end{equation*}
