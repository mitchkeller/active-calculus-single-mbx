\begin{align*} G'(y) =\mathstrut \amp \lim_{h \to 0} \frac{G(y+h) - G(y)}{h}\\ =\mathstrut \amp \lim_{h \to 0} \frac{\sqrt{y+h}-\sqrt{y}}{h}\\ =\mathstrut \amp \lim_{h \to 0} \frac{\sqrt{y+h}-\sqrt{y}}{h} \cdot \frac{\sqrt{y+h} + \sqrt{y}}{\sqrt{y+h} + \sqrt{y}}\\ =\mathstrut \amp \lim_{h \to 0} \frac{(y+h)-y}{h(\sqrt{y+h} + \sqrt{y})}\\ =\mathstrut \amp \lim_{h \to 0} \frac{h}{h(\sqrt{y+h} + \sqrt{y})}\\ =\mathstrut \amp \lim_{h \to 0} \frac{1}{\sqrt{y+h} + \sqrt{y}}\\ =\mathstrut \amp \frac{1}{2\sqrt{y}} \end{align*}
