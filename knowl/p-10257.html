Then let \(u= \csc(3 x)\text{,}\) which means that \(du= -3 \csc(3 x)\cot(3 x) \, dx\text{,}\) and thus \(dx= \frac{du}{-3\csc(3 x)\cot(3 x)}\text{.}\)
\begin{equation*} \begin{aligned} \int 12\cot^{3}\!\left(3x\right)\csc^{3}\!\left(3x\right) \, dx \amp = 12 \int \cot^2(3 x) \csc^2(3 x) \csc(3 x) \cot(3 x) \, dx \\\\ \amp = 12 \int \bigg( \csc^2(3 x)-1 \bigg) \csc^2(3 x) \csc(3 x) \cot(3 x) \, dx \\\\ \amp = 12 \int \bigg( u^2-1 \bigg) u^2 \csc(3 x) \cot(3 x) \frac{du}{-3\csc(3 x)\cot(3 x)} \\\\ \amp = \frac{12}{-3} \int \bigg( u^2-1 \bigg) u^2 \, du \\\\ \amp = \frac{12}{-3} \int \bigg( u^4-u^2 \bigg) \, du \\\\ \amp = \frac{12}{-3} \bigg[\frac{ u^5}{5}-\frac{u^3}{3}\bigg]+C \\\\ \amp = \frac{12}{-3} \bigg[\frac{ \csc^5(3 x)}{5}-\frac{\csc^3(3 x)}{3}\bigg]+C \end{aligned} \end{equation*}
