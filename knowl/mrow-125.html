\begin{align*} f'(x) &= \lim_{h \to 0} \frac{f(x+h)-f(x)}{h}\\ &= \lim_{h \to 0} \frac{\left[\frac{1}{2}(x+h)^2-1\right] - \left[\frac{1}{2}x^2-1\right]}{h}\\ &= \lim_{h \to 0} \frac{\frac{1}{2}\left[x^2+2xh+h^2\right] - \frac{1}{2}\left[x^2\right]}{h}\\ &= \lim_{h \to 0} \frac{1}{2}\frac{2xh+h^2}{h}\\ &= \lim_{h \to 0} x+\frac{h}{2}\\ &= x \end{align*}
