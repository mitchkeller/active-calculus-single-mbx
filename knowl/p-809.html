\begin{equation*} f'(x) = 0\quad\mbox{for}\quad x \lt -4, \end{equation*}
\begin{equation*} f'(x) \lt 0\quad\mbox{for}\quad -4 \lt x \lt 4, \end{equation*}
\begin{equation*} f'(x) \lt 0\quad\mbox{for}\quad 4 \lt x \end{equation*}
