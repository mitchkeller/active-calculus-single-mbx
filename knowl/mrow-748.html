\begin{align*} \int_0^1 \ln(x) \, dx &= \lim_{a \to 0^+} \int_a^1 \ln(x) \, dx\\ &= \lim_{a \to 0^+} \left( x\ln(x)\biggm|_a^1 - \int_{a}^{1} \ dx \right)\\ &= \lim_{a \to 0^+} \left( -a\ln(a) - x\biggm|_{\ln(a)}^{0} \right)\\ &= \lim_{a \to 0^+} \left( -a\ln(a) - (1-a) \right)\\ &= \lim_{a \to 0^+} \left( -a\ln(a) \right) - 1 \end{align*}
