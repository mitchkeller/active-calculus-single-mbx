Then let \(u= \sec(-13 x)\text{,}\) which means that \(du= -13 \sec(-13 x)\tan(-13 x) \, dx\text{,}\) and thus \(dx= \frac{du}{-13\sec(-13 x)\tan(-13 x)}\text{.}\)
\begin{equation*} \begin{aligned} \int 3\tan^{3}\mathopen{}\left(-13x\right)\sec^{3}\mathopen{}\left(-13x\right) \, dx \amp = 3 \int \tan^2(-13 x) \sec^2(-13 x) \sec(-13 x) \tan(-13 x) \, dx \\\\ \amp = 3 \int \bigg( \sec^2(-13 x)-1 \bigg) \sec^2(-13 x) \sec(-13 x) \tan(-13 x) \, dx \\\\ \amp = 3 \int \bigg( u^2-1 \bigg) u^2 \sec(-13 x) \tan(-13 x) \frac{du}{-13\sec(-13 x)\tan(-13 x)} \\\\ \amp = \frac{3}{-13} \int \bigg( u^2-1 \bigg) u^2 \, du \\\\ \amp = \frac{3}{-13} \int \bigg( u^4-u^2 \bigg) \, du \\\\ \amp = \frac{3}{-13} \bigg[\frac{ u^5}{5}-\frac{u^3}{3}\bigg]+C \\\\ \amp = \frac{3}{-13} \bigg[\frac{ \sec^5(-13 x)}{5}-\frac{\sec^3(-13 x)}{3}\bigg]+C \end{aligned} \end{equation*}
