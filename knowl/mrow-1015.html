\begin{align*} \lim_{k \to \infty} \frac{\left| a_{k+1} \right|}{ \left| a_k \right|} \amp = \lim_{k \to \infty} \frac{ \frac{|x|^{k+1}}{(k+1)^2} }{ \frac{| x|^{k}}{k^2} }\\ \amp = \lim_{k \to \infty} |x| \left(\frac{k}{k+1}\right)^2\\ \amp = |x| \lim_{k \to \infty} \left(\frac{k}{k+1}\right)^2\\ \amp = |x| \end{align*}
