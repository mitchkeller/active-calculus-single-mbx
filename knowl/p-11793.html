Then let \(u= \tan(15 x)\text{,}\) which means that \(du= 15 \sec^2(15 x) \, dx\text{,}\) and thus \(dx= \frac{du}{15\sec^2(15 x)}\text{.}\)
\begin{equation*} \begin{aligned} \int 7\tan^{4}\mathopen{}\left(15x\right)\sec^{4}\mathopen{}\left(15x\right) \, dx \amp = 7 \int \tan^4(15 x) \sec^2(15 x) \sec^2(15 x) \, dx \\\\ \amp = 7 \int \tan^4(15 x) \bigg( \tan^2(15 x)+1 \bigg) \sec^2(15 x) \, dx \\\\ \amp = 7 \int u^4\bigg( u^2+1 \bigg) \sec^2(15 x) \frac{du}{15\sec^2(15 x)} \\\\ \amp = \frac{7}{15} \int u^4\bigg( u^2+1 \bigg) \, du \\\\ \amp = \frac{7}{15} \int \bigg( u^6+u^4 \bigg) \, du \\\\ \amp = \frac{7}{15} \bigg[\frac{ u^7}{7}+\frac{u^5}{5}\bigg]+C \\\\ \amp = \frac{7}{15} \bigg[\frac{ \tan^7(15 x)}{7}+\frac{\tan^5(15 x)}{5}\bigg]+C \end{aligned} \end{equation*}
