\begin{align*} \int_e^{\infty} \frac{1}{x\ln(x)} \, dx &= \lim_{b \to \infty} \int_e^{b} \frac{1}{x\ln(x)} \, dx\\ &= \lim_{b \to \infty} \int_1^{\ln(b)} \frac{1}{u} \, du\\ &= \lim_{b \to \infty} \left( \ln(|u|) \right)\biggm|_1^{\ln(b)}\\ &= \lim_{b \to \infty} \left( \ln(|\ln(b)|) - \ln(1) \right)\\ &= \infty \end{align*}
