\begin{equation*} g'(x) = \lim_{h \to 0} \frac{g(x+h)-g(x)}{h} = \lim_{h \to 0} \frac{5\mathopen{}\left(x+h\right)^{2}-1-\left(5x^{2}-1\right)}{h} \end{equation*}
\begin{equation*} = \lim_{h \to 0} \frac{5(x^2+2xh+h^2)-1-5 x^2+1}{h} = \lim_{h \to 0} \frac{10 xh+2 h^2}{h} \end{equation*}
\begin{equation*} = \lim_{h \to 0}(10 x+2h) = 10 x. \end{equation*}
