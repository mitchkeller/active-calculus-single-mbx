\begin{equation*} \frac{dy}{y^2} = (1 + t)\,dt, \end{equation*}
so
\begin{equation*} \int \frac{dy}{y^2} = \int (1 + t)\,dt, \end{equation*}
and
\begin{equation*} -\frac{1}{y} = {1\over2}(1 + t)^2 + C, \end{equation*}
giving
\begin{equation*} y=-\frac{2}{(1 + t)^2 + A}, \end{equation*}
where \(A = 2C\text{.}\) Since \(y=8\) when \(t=1\text{,}\) we have
\begin{equation*} 8 = -\frac{2}{2^2 + A},\quad \hbox{ so } \quad 8 A + 32 = -2, \end{equation*}
and \(A = -{{17\over4}}\text{.}\) Thus
\begin{equation*} y = -\frac{2}{(1 + t)^2 - {{17\over4}}}. \end{equation*}
