\begin{equation*} T(x) = f(a) + \frac{f'(a)}{1!} (x-a) + \frac{f''(a)}{2!} (x-a)^2 + ... + \frac{f^n(a)}{n!} (x-a)^n \end{equation*}
\begin{equation*} f'(x) = e^{x} \end{equation*}
\begin{equation*} f''(x) = e^{x} \end{equation*}
So, in this case,
\begin{equation*} T_2(x) = e^{0.2}+e^{0.2}\!\left(x-0.2\right)+\frac{e^{0.2}}{2}\!\left(x-0.2\right)^{2} \end{equation*}
\begin{equation*} |e^{x} - T_2(x)| = \left|e^{0.2}+e^{0.2}\!\left(-0.7\right)+\frac{e^{0.2}}{2}\cdot 0.49-e^{-0.5}\right| = 0.0591338 \end{equation*}
